def initialize(user)
  @user = user

  if user[:is_admin]
    can :manage, Class, :active=>true, :user_id=>user[:id]
    can :manage, Class2, :foo=>"bar", :user_id=>user[:id]

    can :view, Class do |the_class|
      the_class.active == true && the_class.user_id == user[:id]
    end
  end
end
